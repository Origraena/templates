% meta/functions.tex


%%% COMMANDS

\newcommand{\N}[0]{\mathbb{N}}
\newcommand{\code}[1]{{\em #1}}

\newcommand{\warning}{\bf \color{red}}


\newcommand{\nosection}[1]{
	\refstepcounter{section}
	\addcontentsline{toc}{section}{\protect\numberline{\thesection}#1}
	\markright{#1}
}

\newcommand{\nosubsection}[1]{
	\refstepcounter{subsection}
	\addcontentsline{toc}{subsection}{\protect\numberline{\thesubsection}#1}
	\markright{#1}
}



\newcommand{\theotitle}[2]{
	\vspace{0.5cm}
	\noindent {\bf #1 } {\em #2}\\ 
}

\newcommand{\reftheotitle}[3]{
	\refstepcounter{subsection}
	\addcontentsline{toc}{subsection}{\protect\numberline{\thesubsection}#3}
	\theotitle{#1 \thesubsection}{#2}
}

\newcommand{\definition}[1]{
	\reftheotitle{Définition}{#1}{Définition : #1}
}

\newcommand{\proposition}[1]{
	\reftheotitle{Proposition}{#1}{Proposition}
}

\newcommand{\preuve}[1]{
	\theotitle{Preuve \ref{#1}}{}
}

\newcommand{\observation}[2]{
	\small{
	\theotitle{Observation}{#1}
	{\em #2}
	}
}
